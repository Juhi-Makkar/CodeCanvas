1. Singly Linked List - Node Structure

Fundamental building block with data and pointer to next node.

Node Definition

C struct Node { int data; struct Node* next; };

Insert at Beginning

C void insertAtBegin(struct Node** head, int data) { struct Node* newNode = createNode(data); newNode->next = *head; *head = newNode; }

2. Key Operations

Insert at End

C void insertAtEnd(struct Node** head, int data) { struct Node* newNode = createNode(data); if(*head == NULL) { *head = newNode; return; } struct Node* temp = *head; while(temp->next != NULL) temp = temp->next; temp->next = newNode; }

Delete from Beginning

C void deleteAtBegin(struct Node** head) { if(*head == NULL) return; struct Node* temp = *head; *head = (*head)->next; free(temp); }

Traverse

C void traverse(struct Node* head) { struct Node* temp = head; while(temp != NULL) { printf("%d -> ", temp->data); temp = temp->next; } printf("NULL\n"); }
Time Complexity: Insert/Delete at beginning: O(1), at end/position: O(n)

1. Circular Singly Linked List - Node Structure

Last node points back to first node (head).

Node Definition

C struct Node { int data; struct Node* next; };

Insert at End (Circular)

C void insertAtEnd(struct Node** head, int data) { struct Node* newNode = createNode(data); if(*head == NULL) { *head = newNode; newNode->next = *head; return; } struct Node* temp = *head; while(temp->next != *head) temp = temp->next; temp->next = newNode; newNode->next = *head; }

Traverse Circular

C void traverse(struct Node* head) { if(head == NULL) return; struct Node* temp = head; do { printf("%d -> ", temp->data); temp = temp->next; } while(temp != head); printf(" (back to head)\n"); }

Delete Node (Circular)

C void deleteNode(struct Node** head, int key) { if(*head == NULL) return; struct Node* temp = *head, *prev; if(temp->data == key) { while(temp->next != *head) temp = temp->next; temp->next = (*head)->next; free(*head); *head = temp->next; return; } while(temp != *head && temp->data != key) { prev = temp; temp = temp->next; } if(temp == *head) return; prev->next = temp->next; free(temp); }
Advantage: Continuous traversal without NULL termination

1. Doubly Linked List - Node Structure

Each node has pointers to both next and previous nodes.

Node Definition

C struct DoublyNode { int data; struct DoublyNode* next; struct DoublyNode* prev; };

Insert at Beginning

C void insertAtBegin(struct DoublyNode** head, int data) { struct DoublyNode* newNode = createDoublyNode(data); newNode->next = *head; newNode->prev = NULL; if(*head != NULL) (*head)->prev = newNode; *head = newNode; }

Delete Node (Doubly)

C void deleteNode(struct DoublyNode** head, struct DoublyNode* del) { if(*head == NULL || del == NULL) return; if(*head == del) { *head = del->next; } if(del->next != NULL) del->next->prev = del->prev; if(del->prev != NULL) del->prev->next = del->next; free(del); }
Advantage: Bidirectional traversal, O(1) deletion with node pointer

1. Circular Doubly Linked List - Node Structure

Last node points to first, first points to last. Bidirectional circular.

Node Definition

C struct DoublyNode { int data; struct DoublyNode* next; struct DoublyNode* prev; };

Insert at End (Circular Doubly)

C void insertAtEnd(struct DoublyNode** head, int data) { struct DoublyNode* newNode = createDoublyNode(data); if(*head == NULL) { *head = newNode; newNode->next = newNode; newNode->prev = newNode; return; } struct DoublyNode* last = (*head)->prev; newNode->next = *head; newNode->prev = last; last->next = newNode; (*head)->prev = newNode; }

Traverse Circular Doubly (Forward)

C void traverse(struct DoublyNode* head) { struct DoublyNode* temp = head; if(head == NULL) return; do { printf("%d <-> ", temp->data); temp = temp->next; } while(temp != head); }

Traverse Backward

C void traverseBackward(struct DoublyNode* head) { struct DoublyNode* temp = head->prev; do { printf("%d <- ", temp->data); temp = temp->prev; } while(temp != head->prev); }
Most Powerful: Bidirectional + Circular traversal from any node
