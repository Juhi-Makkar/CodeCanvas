1. Array-Based Stack Structure

Stack implemented using fixed-size array with top pointer.

Stack Structure

C struct Stack { int top; int capacity; int stack[100]; };

Initialize Stack

C s.top = -1; s.capacity = 100;

2. Basic Stack Operations

isEmpty()

C int isEmpty(struct Stack* s) { return (s->top == -1); }

isFull()

C int isFull(struct Stack* s) { return (s->top == s->capacity - 1); }

3. Push & Pop Operations

Push Operation

C void push(struct Stack* s, int data) { if(isFull(s)) return; s->stack[++s->top] = data; }

Pop Operation

C int pop(struct Stack* s) { if(isEmpty(s)) return INT_MIN; return s->stack[s->top--]; }

4. Peek & Size Operations

Peek (Top Element)

C int peek(struct Stack* s) { if(isEmpty(s)) return INT_MIN; return s->stack[s->top]; }

Get Size

C int size(struct Stack* s) { return s->top + 1; }

5. Traversing Stack

Print Stack

C void printStack(struct Stack* s) { for(int i = s->top; i >= 0; i--) { printf("%d ", s->stack[i]); } printf("\n"); }

6. Linked List-Based Stack

Dynamic stack implementation using linked list.

Node Structure

C struct Node { int data; struct Node* next; }; struct Stack { struct Node* top; };

Push (Linked List)

C void push(struct Stack* s, int data) { struct Node* newNode = (struct Node*)malloc(sizeof(struct Node)); newNode->data = data; newNode->next = s->top; s->top = newNode; }

Pop (Linked List)

C int pop(struct Stack* s) { if(s->top == NULL) return INT_MIN; struct Node* temp = s->top; int data = temp->data; s->top = s->top->next; free(temp); return data; }
LIFO: Last In, First Out - Stack follows this principle!

7. Common Stack Applications

Balanced Parentheses

C int isBalanced("()[]{}") { struct Stack s; initStack(&s); for(char c : str) { if(c == '(' || c == '[' || c == '{') push(&s, c); else { if(isEmpty(&s)) return 0; pop(&s); } } return isEmpty(&s); }

Reverse String

C void reverseString(char* str) { struct Stack s; initStack(&s); for(int i = 0; str[i]; i++) push(&s, str[i]); for(int i = 0; str[i]; i++) str[i] = pop(&s); }
