1. Stack Initialization

Creates an empty stack with maximum capacity.

Time: O(1) Space: O(n)

Pseudocode

PROCEDURE initializeStack(maxSize)
    stack[maxSize]
    top = -1
END PROCEDURE

Implementation

C void initStack(int stack[], int *top, int maxSize) { *top = -1; }

2. Push Operation

Adds an element to the top of the stack.

Time: O(1) Space: O(1)

Algorithm Steps

1
Check Overflow
If top == maxSize-1, stack is full
2
Increment Top
top = top + 1
3
Add Element
stack[top] = element

Implementation

C void push(int stack[], int *top, int element, int maxSize) { if(*top == maxSize - 1) { printf("Stack Overflow!\n"); return; } stack[++(*top)] = element; }

3. Pop Operation

Removes and returns the top element from the stack.

Time: O(1) Space: O(1)

Algorithm Steps

1
Check Underflow
If top == -1, stack is empty
2
Return Top
element = stack[top]
3
Decrement Top
top = top - 1

Implementation

C int pop(int stack[], int *top) { if(*top == -1) { printf("Stack Underflow!\n"); return -1; } return stack[(*top)--]; }

4. Peek/Top Operation

Returns the top element without removing it.

Time: O(1) Space: O(1)

Implementation

C int peek(int stack[], int top) { if(top == -1) { printf("Stack is empty!\n"); return -1; } return stack[top]; }

5. Is Empty & Is Full

Check stack conditions.

Time: O(1) Space: O(1)

Implementation

C int isEmpty(int top) { return top == -1; } int isFull(int top, int maxSize) { return top == maxSize - 1; }

6. Infix to Postfix Conversion

Converts infix expression to postfix using stack (Shunting Yard Algorithm).

Time: O(n) Space: O(n)

Algorithm Steps

1
Scan Input
Read each character of infix expression
2
Operand
Directly output to postfix
3
Operator
Pop higher/equal precedence operators first
4
Parentheses
Push '(', pop until '(' for ')'

Implementation

C int precedence(char op) { if(op == '+' || op == '-') return 1; if(op == '*' || op == '/') return 2; return 0; } void infixToPostfix(char infix[], char postfix[]) { int stack[100], top = -1, k = 0; for(int i = 0; infix[i]; i++) { if(infix[i] == '(') stack[++top] = infix[i]; else if(infix[i] == ')') { while(top >= 0 && stack[top] != '(') postfix[k++] = stack[top--]; if(top >= 0) top--; } else if(precedence(infix[i])) { while(top >= 0 && precedence(stack[top]) >= precedence(infix[i])) postfix[k++] = stack[top--]; stack[++top] = infix[i]; } else postfix[k++] = infix[i]; } while(top >= 0) postfix[k++] = stack[top--]; postfix[k] = '\0'; }

7. Postfix Evaluation

Evaluates postfix expression using stack.

Time: O(n) Space: O(n)

Implementation

C int evaluatePostfix(char exp[]) { int stack[100], top = -1; for(int i = 0; exp[i]; i++) { if(isdigit(exp[i])) stack[++top] = exp[i] - '0'; else { int b = stack[top--], a = stack[top--]; if(exp[i] == '+') stack[++top] = a + b; else if(exp[i] == '-') stack[++top] = a - b; else if(exp[i] == '*') stack[++top] = a * b; else if(exp[i] == '/') stack[++top] = a / b; } } return stack[top]; }

8. Inorder Tree Traversal (Iterative)

Performs inorder traversal using stack.

Time: O(n) Space: O(h)

Implementation

C void inorderIterative(Node* root) { Node* curr = root; Stack s; while(curr != NULL || !s.isEmpty()) { while(curr != NULL) { s.push(curr); curr = curr->left; } curr = s.pop(); printf("%d ", curr->data); curr = curr->right; } }

9. Preorder Tree Traversal (Iterative)

Performs preorder traversal using stack.

Time: O(n) Space: O(h)

Implementation

C void preorderIterative(Node* root) { if(!root) return; Stack s; s.push(root); while(!s.isEmpty()) { Node* node = s.pop(); printf("%d ", node->data); if(node->right) s.push(node->right); if(node->left) s.push(node->left); } }

10. Balanced Parentheses

Validates if parentheses are properly balanced.

Time: O(n) Space: O(n)

Implementation

C int isBalanced(char str[]) { int stack[100]; int top = -1; for(int i = 0; str[i]; i++) { if(str[i] == '(' || str[i] == '{' || str[i] == '[') stack[++top] = str[i]; else { if(top == -1) return 0; if(str[i] == ')' && stack[top] != '(') return 0; if(str[i] == '}' && stack[top] != '{') return 0; if(str[i] == ']' && stack[top] != '[') return 0; top--; } } return top == -1; }

11. Next Greater Element

Finds next greater element for each element using stack.

Time: O(n) Space: O(n)

Implementation

C++ vector nextGreater(vector& nums) { int n = nums.size(); vector res(n, -1); stack st; for(int i = n-1; i >= 0; i--) { while(!st.empty() && st.top() <= nums[i]) st.pop(); if(!st.empty()) res[i] = st.top(); st.push(nums[i]); } return res; }

12. Two Stacks in Single Array

Implements two stacks using a single array from both ends.

Time: O(1) Space: O(n)

Implementation

C #define MAX 100 int top1 = -1, top2 = MAX; void push1(int stack[], int x) { if(top1 < top2 - 1) stack[++top1] = x; } void push2(int stack[], int x) { if(top1 < top2 - 1) stack[--top2] = x; } int pop1(int stack[]) { return stack[top1--]; } int pop2(int stack[]) { return stack[top2++]; }

13. Reverse String Using Stack

Reverses a string using stack LIFO property.

Time: O(n) Space: O(n)

Implementation

C void reverseString(char str[]) { int stack[100]; int top = -1, i = 0; // Push all characters while(str[i]) stack[++top] = str[i++]; // Pop and overwrite original string i = 0; while(top >= 0) str[i++] = stack[top--]; str[i] = '\0'; }
