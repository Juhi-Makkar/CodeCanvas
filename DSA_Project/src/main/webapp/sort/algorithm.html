1. Bubble Sort

Repeatedly swap adjacent elements if they are in wrong order.

Time: O(n²) Worst: O(n²) Best: O(n) Space: O(1)

Algorithm Steps

1
Outer Loop
Repeat for n-1 passes
2
Inner Loop
Compare adjacent elements
3
Swap if Needed
Swap if arr[j] > arr[j+1]

Pseudocode

PROCEDURE bubbleSort(arr, n)
FOR i = 0 TO n-2 DO
    FOR j = 0 TO n-i-2 DO
        IF arr[j] > arr[j+1] THEN
            SWAP arr[j], arr[j+1]
        END IF
    END FOR
END FOR
END PROCEDURE

Implementation

C void bubbleSort(int arr[], int n) { for(int i = 0; i < n-1; i++) { for(int j = 0; j < n-i-1; j++) { if(arr[j] > arr[j+1]) { int temp = arr[j]; arr[j] = arr[j+1]; arr[j+1] = temp; } } } }

2. Selection Sort

Select minimum element from unsorted portion and place at beginning.

Time: O(n²) Worst: O(n²) Best: O(n²) Space: O(1)

Algorithm Steps

1
Outer Loop
For each position i from 0 to n-2
2
Find Minimum
Find min in unsorted portion
3
Swap
Swap min with arr[i]

Pseudocode

PROCEDURE selectionSort(arr, n)
FOR i = 0 TO n-2 DO
    min_idx = i
    FOR j = i+1 TO n-1 DO
        IF arr[j] < arr[min_idx] THEN
            min_idx = j
        END IF
    END FOR
    SWAP arr[i], arr[min_idx]
END FOR
END PROCEDURE

Implementation

C void selectionSort(int arr[], int n) { for(int i = 0; i < n-1; i++) { int min_idx = i; for(int j = i+1; j < n; j++) { if(arr[j] < arr[min_idx]) min_idx = j; } int temp = arr[min_idx]; arr[min_idx] = arr[i]; arr[i] = temp; } }

3. Insertion Sort

Build sorted portion one element at a time by inserting into correct position.

Time: O(n²) Worst: O(n²) Best: O(n) Space: O(1) Stable: Yes

Implementation

C void insertionSort(int arr[], int n) { for(int i = 1; i < n; i++) { int key = arr[i]; int j = i - 1; while(j >= 0 && arr[j] > key) { arr[j+1] = arr[j]; j--; } arr[j+1] = key; } }

4. Merge Sort

Divide array into halves, sort them recursively, then merge.

Time: O(n log n) Space: O(n) Stable: Yes

Implementation

C void merge(int arr[], int l, int m, int r) { // Merge two sorted halves } void mergeSort(int arr[], int l, int r) { if(l < r) { int m = l + (r-l)/2; mergeSort(arr, l, m); mergeSort(arr, m+1, r); merge(arr, l, m, r); } }

5. Quick Sort

Partition array around pivot, recursively sort subarrays.

Time: O(n log n) Worst: O(n²) Best: O(n log n) Space: O(log n)

Implementation

C int partition(int arr[], int low, int high) { int pivot = arr[high]; int i = low - 1; for(int j = low; j < high; j++) { if(arr[j] <= pivot) { i++; int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp; } } int temp = arr[i+1]; arr[i+1] = arr[high]; arr[high] = temp; return i+1; } void quickSort(int arr[], int low, int high) { if(low < high) { int pi = partition(arr, low, high); quickSort(arr, low, pi-1); quickSort(arr, pi+1, high); } }

6. Counting Sort

Sort by counting occurrences of each element.

Time: O(n + k) Space: O(k) Stable: Yes

Implementation

C void countingSort(int arr[], int n, int max) { int count[1000] = {0}; int output[n]; // Count occurrences for(int i = 0; i < n; i++) count[arr[i]]++; // Cumulative count for(int i = 1; i <= max; i++) count[i] += count[i-1]; // Build output array for(int i = n-1; i >= 0; i--) output[--count[arr[i]]] = arr[i]; for(int i = 0; i < n; i++) arr[i] = output[i]; }

7. Radix Sort

Sort digits from least to most significant using counting sort.

Time: O(d(n + k)) Space: O(n + k) Stable: Yes

8. Heap Sort

Build max-heap, repeatedly extract max and heapify.

Time: O(n log n) Space: O(1)

Implementation

C void heapify(int arr[], int n, int i) { int largest = i; int l = 2*i + 1; int r = 2*i + 2; if(l < n && arr[l] > arr[largest]) largest = l; if(r < n && arr[r] > arr[largest]) largest = r; if(largest != i) { int temp = arr[i]; arr[i] = arr[largest]; arr[largest] = temp; heapify(arr, n, largest); } } void heapSort(int arr[], int n) { for(int i = n/2-1; i >= 0; i--) heapify(arr, n, i); for(int i = n-1; i >= 0; i--) { int temp = arr[0]; arr[0] = arr[i]; arr[i] = temp; heapify(arr, i, 0); } }

9. Shell Sort

Insertion sort with decreasing gaps between elements.

Time: O(n log n) Space: O(1)

Implementation

C void shellSort(int arr[], int n) { for(int gap = n/2; gap > 0; gap /= 2) { for(int i = gap; i < n; i++) { int temp = arr[i]; int j; for(j = i; j >= gap && arr[j-gap] > temp; j -= gap) arr[j] = arr[j-gap]; arr[j] = temp; } } }
