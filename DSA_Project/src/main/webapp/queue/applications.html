Code Examples

See how queues are implemented in real-world scenarios.

1. CPU Process Scheduler

class ProcessScheduler {
    constructor() {
        this.readyQueue = [];
    }
    
    addProcess(process) {
        this.readyQueue.push(process);
    }
    
    executeNext() {
        return this.readyQueue.shift();
    }
}

2. Print Queue

class PrintQueue {
    constructor() {
        this.jobs = [];
    }
    
    addJob(document) {
        this.jobs.push(document);
    }
    
    printNext() {
        return this.jobs.shift();
    }
}

3. Network Packet Queue

class PacketQueue {
    constructor() {
        this.packets = [];
    }
    
    receivePacket(packet) {
        this.packets.push(packet);
    }
    
    sendNext() {
        return this.packets.shift();
    }
}
