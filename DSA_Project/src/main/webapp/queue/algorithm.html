1. Queue Initialization

Creates an empty queue with maximum capacity.

Time: O(1) Space: O(n)

Pseudocode

PROCEDURE initializeQueue(maxSize)
    queue[maxSize]
    front = -1
    rear = -1
END PROCEDURE

Implementation

C void initQueue(int queue[], int *front, int *rear, int maxSize) { *front = -1; *rear = -1; }

2. Enqueue Operation

Adds an element to the rear of the queue.

Time: O(1) Space: O(1)

Algorithm Steps

1
Check Overflow
If rear == maxSize-1, queue is full
2
First Element
If front == -1, set front = 0
3
Add Element
queue[++rear] = element

Implementation

C void enqueue(int queue[], int *front, int *rear, int element, int maxSize) { if(*rear == maxSize - 1) { printf("Queue Overflow!\\n"); return; } if(*front == -1) *front = 0; queue[++(*rear)] = element; }

3. Dequeue Operation

Removes and returns the front element from the queue.

Time: O(1) Space: O(1)

Algorithm Steps

1
Check Underflow
If front == -1, queue is empty
2
Return Front
element = queue[front]
3
Advance Front
front = front + 1

Implementation

C int dequeue(int queue[], int *front, int *rear) { if(*front == -1) { printf("Queue Underflow!\\n"); return -1; } int element = queue[*front]; if(*front >= *rear) { *front = *rear = -1; } else { (*front)++; } return element; }

4. Front & Rear Operation

Returns front/rear elements without removing them.

Time: O(1) Space: O(1)

Implementation

C int getFront(int queue[], int front) { if(front == -1) { printf("Queue is empty!\\n"); return -1; } return queue[front]; } int getRear(int queue[], int rear) { if(rear == -1) { printf("Queue is empty!\\n"); return -1; } return queue[rear]; }

5. Is Empty & Is Full

Check queue conditions.

Time: O(1) Space: O(1)

Implementation

C int isEmpty(int front, int rear) { return front == -1; } int isFull(int rear, int maxSize) { return rear == maxSize - 1; }

6. Circular Queue Enqueue

Efficient implementation using circular array.

Time: O(1) Space: O(1)

Implementation

C void enqueueCircular(int queue[], int *front, int *rear, int element, int maxSize) { if((*front == 0 && *rear == maxSize-1) || (*rear+1)%maxSize == *front) { printf("Queue Overflow!\\n"); return; } if(*front == -1) *front = *rear = 0; else (*rear) = ((*rear)+1) % maxSize; queue[*rear] = element; }

7. Circular Queue Dequeue

Removes from circular queue efficiently.

Time: O(1) Space: O(1)

Implementation

C int dequeueCircular(int queue[], int *front, int *rear, int maxSize) { if(*front == -1) { printf("Queue Underflow!\\n"); return -1; } int element = queue[*front]; if(*front == *rear) { *front = *rear = -1; } else { *front = (*front + 1) % maxSize; } return element; }

8. Deque Insert Front

Double Ended Queue - insert at front.

Time: O(1) Space: O(1)

Implementation

C void insertFront(int deque[], int *front, int *rear, int element, int maxSize) { if(*front == 0) { printf("Deque Overflow!\\n"); return; } if(*front == -1) { *front = *rear = 0; } else { (*front)--; } deque[*front] = element; }

9. Deque Insert Rear

Double Ended Queue - insert at rear.

Time: O(1) Space: O(1)

Implementation

C void insertRear(int deque[], int *front, int *rear, int element, int maxSize) { if(*rear == maxSize - 1) { printf("Deque Overflow!\\n"); return; } if(*front == -1) { *front = *rear = 0; } else { (*rear)++; } deque[*rear] = element; }

10. BFS Traversal

Breadth First Search using queue.

Time: O(V+E) Space: O(V)

Implementation

C void BFS(int graph[][V], int start, int visited[]) { int queue[100]; int front = 0, rear = -1; visited[start] = 1; enqueue(queue, &front, &rear, start, 100); while(front <= rear) { int curr = dequeue(queue, &front, &rear); printf("%d ", curr); for(int i = 0; i < V; i++) { if(graph[curr][i] && !visited[i]) { visited[i] = 1; enqueue(queue, &front, &rear, i, 100); } } } }

11. Job Scheduling (Round Robin)

Implements Round Robin scheduling using queue.

Time: O(n) Space: O(n)

Implementation

C void roundRobin(Process p[], int n, int timeSlice) { Queue q; int time = 0; for(int i = 0; i < n; i++) enqueue(&q, p[i]); while(!isEmpty(q.front, q.rear)) { Process curr = dequeue(&q); int execTime = min(curr.remainingTime, timeSlice); curr.remainingTime -= execTime; time += execTime; printf("Process %d executed for %d units\\n", curr.id, execTime); if(curr.remainingTime > 0) enqueue(&q, curr); } }

12. Priority Queue Insert

Inserts element based on priority.

Time: O(n) Space: O(1)

Implementation

C void insertPriorityQueue(int pq[], int *front, int *rear, int element, int priority, int maxSize) { if(*rear == maxSize - 1) { printf("Priority Queue Overflow!\\n"); return; } int i = *rear; while(i >= *front && priority > pq[i].priority) { pq[i+1] = pq[i]; i--; } pq[i+1].data = element; pq[i+1].priority = priority; (*rear)++; }

13. Sliding Window Maximum

Finds maximum in each window using deque.

Time: O(n) Space: O(k)

Implementation

C++ vector maxSlidingWindow(vector& nums, int k) { deque dq; vector result; for(int i = 0; i < nums.size(); i++) { while(!dq.empty() && nums[dq.back()] <= nums[i]) dq.pop_back(); dq.push_back(i); if(dq.front() == i-k) dq.pop_front(); if(i >= k-1) result.push_back(nums[dq.front()]); } return result; }
