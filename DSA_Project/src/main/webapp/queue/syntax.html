1. Array Queue Declaration & Initialization

Array-based queue implementation using front and rear pointers.

Structure Declaration

C typedef struct { int data[MAX]; int front, rear; } Queue;

Initialization

C Queue q; q.front = q.rear = -1;
Note: front = rear = -1 indicates empty queue.

2. Basic Operations

Enqueue (Insert)

C void enqueue(Queue* q, int value) { if(q->rear == MAX-1) return; // Overflow if(q->front == -1) q->front = 0; q->data[++q->rear] = value; }

Dequeue (Delete)

C int dequeue(Queue* q) { if(q->front == -1) return -1; // Underflow int val = q->data[q->front++]; if(q->front > q->rear) q->front = q->rear = -1; return val; }

Peek/Front

C int peek(Queue* q) { if(q->front == -1) return -1; return q->data[q->front]; }

3. Status Check Operations

Is Empty

C int isEmpty(Queue* q) { return q->front == -1; }

Is Full

C int isFull(Queue* q) { return q->rear == MAX - 1; }

Size

C int size(Queue* q) { if(q->front == -1) return 0; return q->rear - q->front + 1; }

4. Circular Queue (Optimized)

Circular Enqueue

C void cEnqueue(Queue* q, int value) { if((q->rear + 1) % MAX == q->front) return; if(q->front == -1) q->front = 0; q->data[(++q->rear) % MAX] = value; }

Circular Dequeue

C int cDequeue(Queue* q) { if(q->front == -1) return -1; int val = q->data[q->front]; if(q->front == q->rear) q->front = q->rear = -1; else q->front = (q->front + 1) % MAX; return val; }
Circular Queue: Prevents wastage of space with modulo operation.

5. Linked List Queue

Dynamic queue implementation using pointers. No size limitation.

Node Structure

C typedef struct Node { int data; struct Node* next; } Node;

Queue Structure

C typedef struct { Node* front; Node* rear; } Queue;

Initialization

C Queue q = {NULL, NULL};

6. Linked List Operations

Enqueue

C void enqueue(Queue* q, int value) { Node* newNode = (struct Node*)malloc(sizeof(Node)); newNode->data = value; newNode->next = NULL; if(q->rear == NULL) { q->front = q->rear = newNode; } else { q->rear->next = newNode; q->rear = newNode; } }

Dequeue

C int dequeue(Queue* q) { if(q->front == NULL) return -1; Node* temp = q->front; int val = temp->data; q->front = q->front->next; if(q->front == NULL) q->rear = NULL; free(temp); return val; }

Peek

C int peek(Queue* q) { if(q->front == NULL) return -1; return q->front->data; }

7. Display Queue

Array Queue Display

C void display(Queue q) { for(int i = q.front; i <= q.rear; i++) printf("%d ", q.data[i]); }

Linked List Display

C void display(Queue* q) { Node* temp = q->front; while(temp != NULL) { printf("%d ", temp->data); temp = temp->next; } }
