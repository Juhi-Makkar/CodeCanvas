1. Declaration & Initialization

Learn how to declare and initialize arrays in different programming languages.

Declaration

C int arr[5];

Initialization

C int arr[5] = {1, 2, 3, 4, 5};

Auto Size

C int arr[] = {10, 20, 30}; // Size is 3
Important: In C/C++, size must be known at compile time for static arrays.

2. Accessing & Updating

Read Element

C int x = arr[0]; // First element

Update Element

C arr[0] = 100; // Change to 100

Get Size

C int n = sizeof(arr) / sizeof(arr[0]);
Note: Array indices start from 0. Always check bounds!

3. Traversing an Array

For Loop

C int n = 5; for(int i = 0; i < n; i++) { printf("%d ", arr[i]); }

While Loop

C int i = 0; while(i < n) { printf("%d ", arr[i]); i++; }

4. Insertion Operations

Insert at End

C arr[size] = 50; size++;

Insert at Beginning

C for(int i = size; i > 0; i--) { arr[i] = arr[i-1]; } arr[0] = 99; size++;

Insert at Position

C for(int i = size; i > pos; i--) { arr[i] = arr[i-1]; } arr[pos] = 100; size++;
Time Complexity: O(n)

5. Deletion Operations

Delete from End

C size--;

Delete from Beginning

C for(int i = 0; i < size - 1; i++) { arr[i] = arr[i+1]; } size--;

Delete from Position

C for(int i = pos; i < size - 1; i++) { arr[i] = arr[i+1]; } size--;

6. Searching Operations

Linear Search

C int linearSearch(int arr[], int n, int key) { for(int i = 0; i < n; i++) { if(arr[i] == key) return i; } return -1; }

Binary Search

C int binarySearch(int arr[], int n, int key) { int low = 0, high = n - 1; while(low <= high) { int mid = low + (high - low) / 2; if(arr[mid] == key) return mid; else if(arr[mid] < key) low = mid + 1; else high = mid - 1; } return -1; }
Binary Search: Only works on sorted arrays!

7. Sorting Operations

Bubble Sort

C for(int i = 0; i < n - 1; i++) { for(int j = 0; j < n - i - 1; j++) { if(arr[j] > arr[j+1]) { int temp = arr[j]; arr[j] = arr[j+1]; arr[j+1] = temp; } } }

Selection Sort

C for(int i = 0; i < n - 1; i++) { int minIdx = i; for(int j = i + 1; j < n; j++) { if(arr[j] < arr[minIdx]) minIdx = j; } int temp = arr[minIdx]; arr[minIdx] = arr[i]; arr[i] = temp; }

Insertion Sort

C for(int i = 1; i < n; i++) { int key = arr[i]; int j = i - 1; while(j >= 0 && arr[j] > key) { arr[j+1] = arr[j]; j--; } arr[j+1] = key; }

8. Additional Operations

Find Maximum

C int max = arr[0]; for(int i = 1; i < n; i++) { if(arr[i] > max) max = arr[i]; }

Find Minimum

C int min = arr[0]; for(int i = 1; i < n; i++) { if(arr[i] < min) min = arr[i]; }

Reverse Array

C int start = 0, end = n - 1; while(start < end) { int temp = arr[start]; arr[start] = arr[end]; arr[end] = temp; start++; end--; }

Sum of Elements

C int sum = 0; for(int i = 0; i < n; i++) { sum += arr[i]; }

2D Arrays

Two-dimensional arrays are arrays of arrays, commonly used to represent matrices.

Declaration

C int matrix[3][3];

Initialization

C int matrix[3][3] = {{1,2,3}, {4,5,6}, {7,8,9}};

Access Element

C int x = matrix[0][0]; // First row, first col

Row-wise Traversal

C for(int i = 0; i < 3; i++) { for(int j = 0; j < 3; j++) { printf("%d ", matrix[i][j]); } }

Column-wise Traversal

C for(int j = 0; j < 3; j++) { for(int i = 0; i < 3; i++) { printf("%d ", matrix[i][j]); } }
